Safe tmux via ssh, usage: `tssh <host> <session>`

```
tssh() {
  local host="$1"; shift
  local session="${1:-main}"
  [ $# -gt 0 ] && shift

  command ssh "$host" -t "exec zsh -lc 'tmux new -A -s \"$session\"'"
  local code=$?

  printf '\e[?1000l\e[?1002l\e[?1003l\e[?1006l'
  stty sane 2>/dev/null
  stty echo 2>/dev/null
  return $code
}
```
